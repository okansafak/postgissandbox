## ▶ Data Transformation (Coordinate Conversion)

<RunnableBlock
  sql={`-- Transform UTM 32635 → WGS84 4326
WITH addresses_transformed AS (
  SELECT 
    id,
    street,
    district,
    ST_Transform(
      ST_SetSRID(ST_MakePoint(x_utm, y_utm), 32635),
      4326
    ) AS geom_wgs84
  FROM addresses_staging
)
SELECT 
  id,
  street,
  district,
  ST_X(geom_wgs84) AS longitude,
  ST_Y(geom_wgs84) AS latitude,
  'EPSG:4326' AS crs
FROM addresses_transformed
LIMIT 5;`}
  label="Coordinate transformation"
/>

## ▶ Data Quality Validation

<RunnableBlock
  sql={`-- Quality checks
WITH validation AS (
  SELECT 
    'NULL geometries' AS check_type,
    COUNT(*) FILTER (WHERE geom IS NULL) AS failed_count,
    COUNT(*) AS total_count
  FROM addresses_staging
  UNION ALL
  SELECT 'Valid shapes',
    COUNT(*) FILTER (WHERE NOT ST_IsValid(geom)) AS failed,
    COUNT(*) FROM addresses_staging WHERE geom IS NOT NULL
  UNION ALL
  SELECT 'Duplicates',
    0, COUNT(DISTINCT id) FROM addresses_staging
)
SELECT 
  check_type,
  failed_count,
  ROUND((failed_count::float / total_count * 100)::numeric, 2) AS failure_rate,
  CASE 
    WHEN failed_count = 0 THEN '✓ PASS'
    ELSE '✗ FAIL'
  END AS status
FROM validation;`}
  label="Data quality validation"
/>

## ▶ Incremental Load (CDC)

<RunnableBlock
  sql={`-- Change Data Capture pattern
CREATE TABLE IF NOT EXISTS addresses_log (
  id BIGINT,
  operation VARCHAR(10),
  old_values JSONB,
  new_values JSONB,
  timestamp TIMESTAMP
);

-- Sample CDC events
INSERT INTO addresses_log VALUES
  (1, 'INSERT', NULL, '{"street":"İstiklal"}', NOW()),
  (2, 'UPDATE', '{"district":"Fatih"}', '{"district":"Beyoğlu"}', NOW() + INTERVAL '1 minute'),
  (3, 'DELETE', '{"id":3}', NULL, NOW() + INTERVAL '2 minutes')
ON CONFLICT DO NOTHING;

-- Daily delta summary
SELECT 
  operation,
  COUNT(*) AS count,
  'Today''s changes' AS batch
FROM addresses_log
WHERE timestamp >= CURRENT_DATE
GROUP BY operation;`}
  label="Incremental load (CDC)"
/>
