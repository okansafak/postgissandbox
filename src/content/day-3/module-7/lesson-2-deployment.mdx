## ▶ Connection Pooling (pgBouncer)

<RunnableBlock
  sql={`-- Connection pool optimization

  'pgBouncer' AS pooler,
  'Transaction mode' AS pool_mode,
  1000 AS max_client_connections,

  20 AS min_pool_size,
  '99% CPU overhead reduction' AS benefit

-- Pool hit ratio hedefi: > 95%

  0.97::numeric(3,2) AS pool_hit_ratio,
  CASE 
    WHEN 0.97 > 0.95 THEN '✓ Optimal'
    ELSE '✗ Needs tuning'
  END AS status;`}
