## ▶ Delivery Route Optimization

<RunnableBlock
  sql={`-- Order-to-vehicle assignment (greedy insertion)
CREATE TABLE IF NOT EXISTS orders (
  id INT PRIMARY KEY,
  location GEOMETRY(POINT, 4326),
  time_window_start TIME,
  time_window_end TIME,
  weight_kg NUMERIC
);

CREATE TABLE IF NOT EXISTS vehicles (
  id INT PRIMARY KEY,
  current_location GEOMETRY(POINT, 4326),
  capacity_kg INT,
  current_load_kg INT
);

-- Sample: Assign 10 orders to 3 vehicles
INSERT INTO orders VALUES
  (1, ST_MakePoint(28.97, 41.00), '09:00', '12:00', 5),
  (2, ST_MakePoint(28.98, 41.01), '10:00', '14:00', 3),
  (3, ST_MakePoint(28.96, 41.02), '11:00', '15:00', 4)
ON CONFLICT DO NOTHING;

INSERT INTO vehicles VALUES
  (1, ST_MakePoint(28.95, 40.99), 50, 10),
  (2, ST_MakePoint(28.99, 41.03), 50, 15)
ON CONFLICT DO NOTHING;

-- Assignment result
WITH feasible AS (
  SELECT 
    o.id AS order_id,
    v.id AS vehicle_id,
    ST_Distance(v.current_location::geography, o.location::geography)::int4 AS distance_m
  FROM orders o, vehicles v
  WHERE v.current_load_kg + o.weight_kg <= v.capacity_kg
)
SELECT 
  order_id,
  vehicle_id,
  distance_m,
  ROW_NUMBER() OVER (PARTITION BY order_id ORDER BY distance_m) AS rank
FROM feasible
WHERE ROW_NUMBER() OVER (PARTITION BY order_id ORDER BY distance_m) <= 1;`}
  label="Route optimization (greedy insertion)"
/>

## ▶ Demand Forecasting

<RunnableBlock
  sql={`-- 7-day demand forecast
CREATE TABLE IF NOT EXISTS daily_deliveries (
  date DATE,
  hour INT,
  delivery_count INT
);

INSERT INTO daily_deliveries VALUES
  ('2024-01-01', 9, 45),
  ('2024-01-01', 10, 67),
  ('2024-01-01', 11, 89),
  ('2024-01-01', 12, 120),
  ('2024-01-02', 9, 48),
  ('2024-01-02', 10, 70)
ON CONFLICT DO NOTHING;

-- Forecast next 7 days (simple moving average)
WITH daily_avg AS (
  SELECT 
    hour,
    AVG(delivery_count)::int4 AS avg_deliveries,
    STDDEV(delivery_count)::numeric(6,2) AS std_dev
  FROM daily_deliveries
  GROUP BY hour
)
SELECT 
  hour,
  avg_deliveries AS forecast_count,
  ROUND((avg_deliveries * 1.15)::numeric)::int4 AS with_15pct_buffer,
  CEILING(avg_deliveries::numeric / 5)::int4 AS vehicles_needed
FROM daily_avg
ORDER BY hour;`}
  label="Demand forecasting (7 days)"
/>

## ▶ Anomaly Detection

<RunnableBlock
  sql={`-- Real-time anomaly detection (LOF-like)
WITH hourly_stats AS (
  SELECT 
    EXTRACT(HOUR FROM CURRENT_TIMESTAMP)::int4 AS hour,
    COUNT(*) AS current_count,
    (SELECT AVG(delivery_count) FROM daily_deliveries 
     WHERE EXTRACT(HOUR FROM date) = EXTRACT(HOUR FROM CURRENT_TIMESTAMP)) AS expected_count
)
SELECT 
  hour,
  current_count,
  expected_count::int4,
  CASE 
    WHEN expected_count IS NULL THEN 'NO_BASELINE'
    WHEN current_count > expected_count * 1.5 THEN '🔴 ANOMALY_HIGH'
    WHEN current_count < expected_count * 0.5 THEN '🔴 ANOMALY_LOW'
    ELSE '🟢 NORMAL'
  END AS status
FROM hourly_stats;`}
  label="Real-time anomaly detection"
/>
