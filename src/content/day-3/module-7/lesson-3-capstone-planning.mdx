<RunnableBlock

CREATE TABLE IF NOT EXISTS geographic_features (
  id BIGSERIAL PRIMARY KEY,
  name VARCHAR(255),
  geom GEOMETRY(POINT, 4326),
  feature_type VARCHAR(50),
  domain VARCHAR(50),
  attributes JSONB
);

SELECT COUNT(*) AS feature_count, domain FROM geographic_features GROUP BY domain;`}
